const hub=new THREE.Mesh(new THREE.CylinderGeometry(.08,.08,.08,8),DARK);G.add(hub); // Batterie const batt=new THREE.Mesh(new THREE.BoxGeometry(.18,.08,.13),BATT);batt.position.y=.07;G.add(batt); // 4 Moteurs + Hélices + CROIX BLANCHE à chaque extrémité G.userData.rotors=[]; [[.36,0,0],[-.36,0,0],[0,0,.36],[0,0,-.36]].forEach(([mx,my,mz],idx)=>{ // Moteur const mot=new THREE.Mesh(new THREE.CylinderGeometry(.036,.042,.055,8),DARK); mot.position.set(mx,.028,mz);G.add(mot); // ── CROIX BLANCHE visible à chaque extrémité ── const cx1=new THREE.Mesh(new THREE.BoxGeometry(.16,.014,.030),WHT); cx1.position.set(mx,.070,mz);G.add(cx1); const cx2=new THREE.Mesh(new THREE.BoxGeometry(.030,.014,.16),WHT); cx2.position.set(mx,.070,mz);G.add(cx2); // Anneau orange sur moteur const ring=new THREE.Mesh(new THREE.TorusGeometry(.044,.011,6,14),ARM); ring.rotation.x=Math.PI/2;ring.position.set(mx,.062,mz);G.add(ring); }); G.add(RG); // ── ANTENNE + LED rouge (visible de loin) ── const ant=new THREE.Mesh(new THREE.CylinderGeometry(.005,.005,.30,5),DARK); ant.position.set(.07,.24,0);G.add(ant); const led=new THREE.Mesh(new THREE.SphereGeometry(.028,7,5),GLOW);
